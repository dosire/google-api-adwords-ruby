    def initialize(filename, log_to_console=false)
      @filename = filename
      @loggers = []
      if log_to_console
        stderr_logger = Logger.new(STDERR)
        stderr_logger.level = Logger::INFO
        @loggers << stderr_logger
      end
    end

    def log_to_file(path='.')
